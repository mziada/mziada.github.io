from random import randint

rock = '''
    _______
---'   ____)
      (_____)
      (_____)
      (____)
---.__(___)
'''

paper = '''
    _______
---'   ____)____
          ______)
          _______)
         _______)
---.__________)
'''

scissors = '''
    _______
---'   ____)____
          ______)
       __________)
      (____)
---.__(___)
'''

game_images = [rock, paper, scissors]

def play_game():
    # Get user choice
    user_choice = int(Element("user-choice").value)
    
    # Display user's choice image
    user_image_element = Element("user-image")
    user_image_element.write(game_images[user_choice])
    
    # Generate computer's choice
    computer_choice = randint(0, 2)
    
    # Display computer's choice image
    computer_image_element = Element("computer-image")
    computer_image_element.write(game_images[computer_choice])
    
    # Determine the result
    result_element = Element("result")
    if user_choice == computer_choice:
        result_element.write("It's a Draw!")
    elif (user_choice == 0 and computer_choice == 2) or \
         (user_choice == 1 and computer_choice == 0) or \
         (user_choice == 2 and computer_choice == 1):
        result_element.write("You Win!")
    else:
        result_element.write("You Lose!")
